## Creating a VM with existing root disk
Step-1


NEW_MAC=$(printf '52:54:00:%02x:%02x:%02x\n' $[RANDOM%256] $[RANDOM%256] $[RANDOM%256])

# 2. Print it to your screen so you have a copy of it
echo "Your new VM MAC address is: $NEW_MAC"

# 3. Register this MAC to your 10.10.1.100 IP in the default network
sudo virsh net-update default add ip-dhcp-host \
  "<host mac='$NEW_MAC' ip='10.10.1.100'/>" \
  --live --config

# Building the VM:

sudo virt-install \
  --name controller.lab.example.com \
  --ram 2048 \
  --vcpus 2 \
  --os-variant almalinux10 \
  --disk path=/kvm-storage/images/controller-root.qcow2,format=qcow2 \
  --network network=default,mac=$NEW_MAC \
  --network network=host-only \
  --import \
  --noautoconsole
